_بعد:_
```
const wss = new WebSocketServer({ server, path: '/ws/live', verifyClient: (info, cb) => {
  const token = new URL(info.req.url, 'http://localhost').searchParams.get('token');
  if (token === process.env.WS_SECRET) { cb(true); } else { cb(false, 401, 'Unauthorized'); }
}});
```
