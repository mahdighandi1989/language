```
const metrics = {
  totalRequests: 0,
  errorCount: 0,
  totalLatency: 0,
  requestCount: 0,
  endpoints: {}
};

export function metricsMiddleware(req, res, next) {
  const start = Date.now();
  
  res.on('finish', () => {
    const latency = Date.now() - start;
    metrics.totalRequests++;
    metrics.totalLatency += latency;
    metrics.requestCount++;
    
    if (res.statusCode >= 400) {
      metrics.errorCount++;
    }
    
    const path = req.path;
    if (!metrics.endpoints[path]) {
      metrics.endpoints[path] = { count: 0, errors: 0, totalLatency: 0 };
    }
    metrics.endpoints[path].count++;
    metrics.endpoints[path].totalLatency += latency;
    if (res.statusCode >= 400) {
      metrics.endpoints[path].errors++;
    }
  });
  
  next();
}

export function getMetrics(format) {
  const avgLatency = metrics.requestCount > 0 ? (metrics.totalLatency / metrics.requestCount).toFixed(2) : 0;
  
  if (format === 'prometheus') {
    return `# HELP http_requests_total Total number of HTTP requests
# TYPE http_requests_total counter
http_requests_total ${metrics.totalRequests}
# HELP http_errors_total Total number of HTTP errors (status >= 400)
# TYPE http_errors_total counter
http_errors_total ${metrics.errorCount}
# HELP http_request_duration_ms Average request duration in milliseconds
# TYPE http_request_duration_ms gauge
http_request_duration_ms ${avgLatency}
`;
  }
  
  return {
    totalRequests: metrics.totalRequests,
    errorCount: metrics.errorCount,
    averageLatencyMs: parseFloat(avgLatency),
    endpoints: metrics.endpoints
  };
}
```
